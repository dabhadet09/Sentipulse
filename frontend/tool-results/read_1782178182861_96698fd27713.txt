    51→### API Contracts (for subagents):
    52→- GET /api/analyses → { analyses: [{id, platform, contentType, title, overallSentiment, sentimentScore, itemCount, isPremium, createdAt}], stats: {total, positive, negative, neutral, mixed, byPlatform:{youtube,reddit,x,instagram}} }
    53→- GET /api/analyses/[id] → { analysis: {...fields, results: AnalysisResult, contentItems: AnalysisItem[]} }
    54→- POST /api/analyze body: { platform, contentType, title, sourceUrl?, items?: [{text, author?}], sampleId? } → { success, analysisId, results: AnalysisResult }
    55→- GET /api/me → { user: {id,name,email,role,subscriptionTier,bio,avatar,createdAt}, analysisCount, applicationCount }
    56→- PATCH /api/me body: { name?, bio? }
    57→- GET /api/premium/apply → { applications: [{id, reason, platform, useCase, status, adminNote, createdAt, reviewedAt}] }
    58→- POST /api/premium/apply body: { reason, platform, useCase }
    59→- GET /api/admin/stats → { totals:{users,premiumUsers,adminUsers,analyses,premiumAnalyses,pendingApplications,approvedApplications,rejectedApplications}, platformBreakdown:[{platform,count}], sentimentBreakdown:[{sentiment,count}], dailyNewUsers:[{date,count}], recentActivity:[{action,detail,createdAt,user:{name,email}}] }
    60→- GET /api/admin/users → { users: [{id,name,email,role,subscriptionTier,createdAt,bio,_count:{analyses,applications,activityLogs}}] }
    61→- PATCH /api/admin/users body: { userId, action: "make_admin"|"remove_admin"|"grant_premium"|"revoke_premium" }
    62→- DELETE /api/admin/users?userId=...
    63→- GET /api/admin/applications?status=PENDING|APPROVED|REJECTED|all → { applications: [{id, reason, platform, useCase, status, adminNote, createdAt, reviewedAt, user:{id,name,email,subscriptionTier,createdAt,_count:{analyses}}}] }
    64→- PATCH /api/admin/applications body: { applicationId, action: "approve"|"reject", adminNote? }
    65→- POST /api/livestream/analyze body: { messages?: [{text,author?}], segment?: number, title?, platform? } → { success, segment, results: AnalysisResult, timestamp }
